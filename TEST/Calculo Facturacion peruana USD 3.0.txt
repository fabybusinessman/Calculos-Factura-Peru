    <script>
        // Función para realizar los cálculos
        function calculateInvoice() {
            // Obtener el valor del tipo de cambio del campo de entrada
            const exchangeRateInput = document.getElementById('exchangeRate');
            const exchangeRateUSDToPEN = parseFloat(exchangeRateInput.value);

            // Obtener el valor de venta del campo de entrada
            const salesValueUSDInput = document.getElementById('salesValueUSD');
            const salesValueUSD = parseFloat(salesValueUSDInput.value);

            // Validar si los valores ingresados son números válidos y positivos
            if (isNaN(exchangeRateUSDToPEN) || exchangeRateUSDToPEN <= 0) {
                 alert('Por favor, ingrese un valor numérico válido y positivo para el Tipo de cambio.');
                 // Limpiar campos de resultado si el valor no es válido
                 document.getElementById('igvValue').innerText = '--';
                 document.getElementById('totalAmountUSD').innerText = '--';
                 document.getElementById('tractionValuePEN').innerText = '--';
                 document.getElementById('creditValueUSD').innerText = '--';
                 return; // Detener la ejecución si el valor no es válido
            }

            if (isNaN(salesValueUSD) || salesValueUSD < 0) {
                alert('Por favor, ingrese un valor numérico válido y positivo para el Valor de la Venta.');
                // Limpiar campos de resultado si el valor no es válido
                document.getElementById('igvValue').innerText = '--';
                document.getElementById('totalAmountUSD').innerText = '--';
                document.getElementById('tractionValuePEN').innerText = '--';
                document.getElementById('creditValueUSD').innerText = '--';
                return; // Detener la ejecución si el valor no es válido
            }


            // Calcular IGV (18% del valor de venta en USD)
            const igvValueUSD = salesValueUSD * 0.18;

            // Calcular el Importe Total en USD (Valor de Venta + IGV)
            const totalAmountUSD = salesValueUSD + igvValueUSD;

            // Convertir el Importe Total a PEN (base para la tracción)
            const totalAmountPEN = totalAmountUSD * exchangeRateUSDToPEN;

            // Calcular el porcentaje de tracción (12% del Importe Total en PEN)
            // Este valor se redondea para mostrarlo
            let tractionValuePEN_rounded = totalAmountPEN * 0.12;
            tractionValuePEN_rounded = Math.round(tractionValuePEN_rounded);

            // Calcular el Crédito (88% del Importe Total en USD)
            const creditValueUSD = totalAmountUSD * 0.88;

            // Mostrar los resultados en los elementos span
            document.getElementById('igvValue').innerText = igvValueUSD.toFixed(2) + ' USD';
            document.getElementById('totalAmountUSD').innerText = totalAmountUSD.toFixed(2) + ' USD'; // Mostrar el importe total
            document.getElementById('tractionValuePEN').innerText = tractionValuePEN_rounded.toFixed(0) + ' PEN'; // Mostrar la tracción redondeada sin decimales
            document.getElementById('creditValueUSD').innerText = creditValueUSD.toFixed(2) + ' USD'; // Mostrar el valor del crédito con 2 decimales y etiqueta USD
        }

        // No ejecutar el cálculo al cargar la página inicialmente, esperamos la entrada del usuario
        // window.onload = calculateInvoice; // Comentado para no calcular automáticamente al cargar
    </script>
